4.4 Сервис пользователей (app/services/user_service.py) 
 
import secrets 
import string 
from sqlalchemy import select 
from app.core.database import async_session 
from app.models.user import User 
 
class UserService: 
async def get_or_create_user(self, telegram_id: int, username: str = None, 
first_name: str = None, last_name: str = None, 
referral_code: str = None): 
async with async_session() as session: 
result = await session.execute( 
select(User).where(User.telegram_id == telegram_id) 
) 
user = result.scalar_one_or_none() 
        if user: 
            # Обновляем данные 
            user.username = username 
            user.first_name = first_name 
            user.last_name = last_name 
            await session.commit() 
            return user 
         
        # Создаём нового пользователя 
        new_referral_code = self._generate_referral_code() 
         
        # Проверяем реферера 
        referrer_id = None 
        if referral_code: 
            referrer_result = await session.execute( 
                select(User).where(User.referral_code == referral_code) 
            ) 
            referrer = referrer_result.scalar_one_or_none() 
            if referrer: 
                referrer_id = referrer.id 
         
        user = User( 
            telegram_id=telegram_id, 
            username=username, 
            first_name=first_name, 
            last_name=last_name, 
            referral_code=new_referral_code, 
            referred_by=referrer_id 
        ) 
        session.add(user) 
        await session.commit() 
        await session.refresh(user) 
        return user 
 
def _generate_referral_code(self) -> str: 
    alphabet = string.ascii_letters + string.digits 
    return ''.join(secrets.choice(alphabet) for _ in range(8))

import httpx 
from app.core.config import settings 
 
class MarzbanService: 
def init(self): 
self.base_url = settings.MARZBAN_URL.rstrip('/') 
self.username = settings.MARZBAN_USERNAME 
self.password = settings.MARZBAN_PASSWORD 
self.token = None 
async def _get_token(self): 
    if self.token: 
        return self.token 
     
    async with httpx.AsyncClient(verify=not settings.MARZBAN_INSECURE) as client: 
        response = await client.post( 
            f"{self.base_url}/api/admin/token", 
            data={"username": self.username, "password": self.password} 
        ) 
        response.raise_for_status() 
        self.token = response.json()["access_token"] 
        return self.token 
 
async def create_user(self, username: str, expire_days: int = 30) -> dict: 
    token = await self._get_token() 
     
    async with httpx.AsyncClient(verify=not settings.MARZBAN_INSECURE) as client: 
        response = await client.post( 
            f"{self.base_url}/api/user", 
            headers={"Authorization": f"Bearer {token}"}, 
            json={ 
                "username": username, 
                "expire": expire_days * 86400,  # секунды 
                "data_limit": 0,  # безлимит 
                "proxies": { 
                    "vless": {}, 
                    "vmess": {} 
                }, 
                "inbounds": { 
                    "vless": ["VLESS Reality TCP"], 
                    "vmess": ["VMess TCP"] 
                } 
            } 
        ) 
        response.raise_for_status() 
        return response.json() 
 
async def get_user_config(self, username: str) -> dict: 
    token = await self._get_token() 
     
    async with httpx.AsyncClient(verify=not settings.MARZBAN_INSECURE) as client: 
        response = await client.get( 
            f"{self.base_url}/api/user/{username}", 
            headers={"Authorization": f"Bearer {token}"} 
        ) 
        response.raise_for_status() 
        return response.json() 
 
async def disable_user(self, username: str): 
    token = await self._get_token() 
     
    async with httpx.AsyncClient(verify=not settings.MARZBAN_INSECURE) as client: 
        response = await client.put( 
            f"{self.base_url}/api/user/{username}", 
            headers={"Authorization": f"Bearer {token}"}, 
            json={"status": "disabled"} 
        ) 
        response.raise_for_status() 
        return response.json()

import qrcode 
from io import BytesIO 
from aiogram.types import BufferedInputFile 
 
def generate_qr_code(data: str) -> BufferedInputFile: 
qr = qrcode.QRCode(version=1, box_size=10, border=5) 
qr.add_data(data) 
qr.make(fit=True) 
img = qr.make_image(fill_color="black", back_color="white") 
buffer = BytesIO() 
img.save(buffer, format='PNG') 
buffer.seek(0) 
 
return BufferedInputFile(buffer.read(), filename="qr_code.png")

from datetime import datetime, timedelta 
from sqlalchemy import select, and_ 
from app.core.database import async_session 
from app.models.subscription import Subscription 
from app.models.user import User 
from app.bot.main import bot 
class NotificationService: 
async def notify_expiring_subscriptions(self): 
async with async_session() as session: 
now = datetime.utcnow() 
        # Уведомление за 7 дней 
        seven_days = now + timedelta(days=7) 
        result = await session.execute( 
            select(Subscription, User).join(User).where( 
                and_( 
                    Subscription.end_date <= seven_days, 
                    Subscription.end_date > now, 
                    Subscription.status == "active" 
                ) 
            ) 
        ) 
         
        for sub, user in result: 
            days_left = (sub.end_date - now).days 
             
            if days_left == 7: 
                await bot.send_message( 
                    user.telegram_id, 
                    "
⚠
 Ваша подписка заканчивается через 7 дней!\n\n" 
                    "Продлите сейчас, чтобы не потерять доступ." 
                ) 
            elif days_left == 3: 
                await bot.send_message( 
                    user.telegram_id, 
                    "
⏳
 Осталось 3 дня до окончания подписки!\n\n" 
                    "Не забудьте продлить." 
                ) 
            elif days_left == 1: 
                await bot.send_message( 
                    user.telegram_id, 
                    "
🚨
 Завтра отключение!\n\n" 
                    "Срочно продлите подписку."

security = HTTPBasic() 
def verify_admin(credentials: HTTPBasicCredentials = Depends(security)): 
is_correct_username = secrets.compare_digest(credentials.username, 
settings.ADMIN_USERNAME) 
is_correct_password = secrets.compare_digest(credentials.password, 
settings.ADMIN_PASSWORD) 
if not (is_correct_username and is_correct_password): 
raise HTTPException(status_code=401, detail="Unauthorized") 
return credentials.username 
@admin_app.get("/")

) 
return templates.TemplateResponse("dashboard.html", {

"total_users": total_users.scalar(), 
"total_revenue": total_payments.scalar() or 0, 
"active_subscriptions": active_subs.scalar() 
})
